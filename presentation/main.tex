\section{Contribution}
% ==========================================================

\begin{frame}{Our contribution, in one slide}
  \small
  \begin{enumerate}
    \item \textbf{A strong frozen baseline.} SIGReg-ambient JEPA matches the LeJEPA/Laya recipe's value on our setup (pipeline-helped: organizers' preprocessing $+$ \texttt{eb\_jepa} dataloader/losses) -- not a controlled head-to-head with Laya.
    \item \textbf{A broad ablation that does not beat it.} Ambient/tangent SIGReg, ambient/tangent PEIRA, Graph-JEPA, Fourier-JEPA -- none improves balanced accuracy beyond noise ($\leq 0.013$). Ambient SIGReg is hard to beat.
    \item \textbf{Latent geometry is the methods contribution.} Our latent separates normal/abnormal $\sim$2$\times$ better than random on TUAB; we visualise it with \textbf{de Surrel's AIRM Riemannian t-SNE} -- the principled metric for SPD/EEG covariances -- which gives a small, \emph{consistent} edge over classical t-SNE \emph{where cluster structure exists}. Tangent training yields the \emph{best} covariance-latent silhouette (TUAB \& TUSZ), though \emph{not} better balanced accuracy -- the geometry objective sharpens the geometry, not the task.
    \item \textbf{Rigor over spin.} We report the limits: \emph{decodable $\ne$ clustered} (TUEV), \emph{sustained vs transient}, the AIRM edge within noise. An honest negative beats a fragile positive.
  \end{enumerate}
\end{frame}

\section{Motivation}
% ==========================================================

\begin{frame}{The question we actually asked}
  \vfill
  \begin{itemize}
    \item \textbf{EEG foundation models} are everywhere -- but most report \emph{fine-tuned} numbers. What survives when you \textbf{freeze} the encoder?
    \item \textbf{Self-supervised JEPA} avoids collapse with an \emph{anti-collapse regulariser}. The EEG covariance lives on a curved \textbf{SPD manifold}, not in flat Euclidean space.
    \item \textbf{Our pre-registered hypothesis}: making anti-collapse \emph{geometry-aware} (act in the SPD \textbf{tangent}) and \emph{distribution-free} (\textbf{PEIRA}) beats a plain ambient baseline.
    \item \textbf{What we ship}: a controlled study with an \emph{honest} answer -- a strong frozen probe, a clean negative result with a geometric mechanism, and label/data-efficiency curves.
  \end{itemize}
  \vfill
  \begin{center}\small
    \textit{We are not a foundation model, not a world model, not SOTA in 24h.}
  \end{center}
\end{frame}

  \end{itemize}

\section{Data}
% ==========================================================

\begin{frame}{Data: TUH Abnormal EEG (TUAB)}
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{itemize}
      \item \textbf{Task}: binary \emph{normal} vs \emph{abnormal} EEG, recording level.
      \item \textbf{Signal}: \nchan{} channels @ \sfreq~Hz; we sample \winsec~s windows ($\winsamp$ samples), per-channel z-scored.
      \item \textbf{Standard split}: \nrectr{} train / \nrecev{} eval recordings, \textbf{patient-disjoint} (\npattr{} vs \npatev{} patients, overlap $=0$ -- verified).
      \item \textbf{Pretraining}: SSL reads only \emph{unlabeled} train recordings; labels enter at the probe alone.
    \end{itemize}
    \column{0.44\textwidth}
    \begin{block}{Why we don't chase accuracy}
      \footnotesize
      TUAB labels come from clinical reports and are known to be noisy; we are not aware of a published label-noise / Bayes-ceiling estimate.
      \textbf{Sub-1-point gaps without multi-seed CIs are not meaningful} -- so we report
      \emph{balanced accuracy on the full split}, 3 seeds, with error bars.
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Where can anti-collapse act? Ambient vs SPD-tangent}
  \vspace{0.2em}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Ambient (Euclidean).}\\
    Pool the feature map over time:
    \[ \mathbf{z} = \tfrac{1}{T'}\textstyle\sum_t f_\theta(x)_{:,t} \in \mathbb{R}^{\dmodel}. \]
    Regularise directly in $\mathbb{R}^{\dmodel}$. This is the \textbf{Laya-like} baseline.

    \column{0.5\textwidth}
    \textbf{Tangent (SPD log-Euclidean).}\\
    Per-window temporal covariance of a $\dcov$-dim projection:
    \[ C = \tfrac{1}{T'-1}\tilde F\tilde F^\top \in \mathrm{SPD}(\dcov), \]
    \[ \mathbf{t} = \mathrm{vec}_{\sqrt 2}\big(\log_m C\big)\in\mathbb{R}^{\dcov(\dcov+1)/2}. \]
    Regularise in the tangent at the identity.
  \end{columns}

  \vspace{0.6em}
  \begin{alertblock}{The clean knob}
    \footnotesize The probe reads the \emph{same} pooled representation in both cases. Only the \emph{space where anti-collapse is enforced} changes -- this isolates the effect of \textbf{geometry}, nothing else.
  \end{alertblock}
\end{frame}

\begin{frame}{The ladder -- a $2\times2$ ($+$ reference) factorial}
  \vspace{0.4em}
  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    \textbf{regulariser} $\backslash$ \textbf{space} & \textbf{ambient} (Euclidean) & \textbf{tangent} (SPD) \\
    \midrule
    \textbf{VICReg} (variance/cov) & \texttt{C0} reference & --- \\
    \textbf{SIGReg} (Gaussianity slices) & \texttt{C1} \emph{Laya-like baseline} & \texttt{C2} geometry-aware \\
    \textbf{PEIRA} (distribution-free) & \texttt{C3} & \texttt{C4} \emph{ex-hypothesis} \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.8em}
  \begin{itemize}
    \item \textbf{SIGReg} (LeJEPA/BCS): tests random 1-D slices for Gaussianity -- assumes an \emph{isotropic} target.
    \item \textbf{PEIRA}: distribution-free anti-collapse -- no target distribution to mis-specify.
    \item \textbf{Hypothesis}: the geometry-aware / distribution-free cells (\texttt{C2}, \texttt{C4}) beat the ambient baseline \texttt{C1}.
  \end{itemize}
\end{frame}

\begin{frame}{Collapse is the \#1 risk -- so we watched it}
  \vspace{0.4em}
  \begin{columns}[T]
    \column{0.55\textwidth}
    A degenerate encoder maps every window to (almost) the same vector. We log three diagnostics on each batch:
    \begin{itemize}
      \item \textbf{Effective rank} -- entropy of the singular-value spectrum; $\to 1$ means collapse.
      \item \textbf{Per-dim std} -- $\to 0$ means collapse.
      \item \textbf{Off-diagonal covariance} -- redundancy across dimensions.
    \end{itemize}
    \column{0.45\textwidth}
    \begin{block}{Rule}
      \footnotesize Kill flat runs early. The fix for collapse is \emph{more} regularisation, not less.
      Every reported cell was checked non-degenerate (high eff.\ rank; PEIRA's $\mathrm{tr}_P$ rising).
    \end{block}
    \begin{exampleblock}{Jury asset}
      \footnotesize ``We didn't just avoid collapse -- we \emph{visualised} it.''
    \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Apples-to-apples: it is not an in-domain artifact}
  \vspace{0.4em}
  \begin{itemize}
    \item \textit{``You pretrained on TUAB, the FMs didn't''} -- the obvious attack on the in-domain number. So we ran the fair version.
    \item Pretrain SIGReg on a \textbf{general TUH corpus} (\textbf{TUSZ} seizure; TUAB-eval patients \emph{excluded} $\Rightarrow$ patient-disjoint), freeze, linear-probe TUAB -- the FMs' own regime (general pretrain $\to$ frozen TUAB).
    \item \textbf{Result: 0.814 BA / 0.889 AUROC} -- essentially equal to in-domain \ba{}. \textbf{The in-domain advantage is \emph{not} the driver.}
  \end{itemize}
  \vspace{0.3em}
  \begin{alertblock}{Honest caveat}
    \footnotesize TUSZ is general \emph{same-site} TUH (not multi-site like the FMs); single seed. We say ``match while frozen'', never ``beat''.
  \end{alertblock}
\end{frame}

\begin{frame}{The honest denominator: a random encoder already scores \floorv{}}
  \vspace{0.3em}
  \begin{itemize}
    \item In our pipeline, a \textbf{random (untrained) conv encoder} + linear probe reaches $\sim$\floorv{} BalAcc.
    \item EEG abnormality is \emph{power-driven}; random conv filters preserve band power. The strong frozen numbers are largely \emph{architecture $+$ task}.
    \item This floor is $\approx$ LaBraM-linear (0.795) / EEGPT-frozen (0.798) \emph{as self-reported by those papers} (different protocols); under the one apples-to-apples bench we cite (EEG-FM-Bench) the floor sits \emph{above} all frozen FMs.
  \end{itemize}
  \vspace{0.4em}
  \begin{alertblock}{Why disclose it?}
    \footnotesize It pre-empts ``is the SSL doing anything?'', reframes our contribution as a \textbf{quantified} SSL increment ($\sim{+}0.04$ over random, $\sim4\times$ label efficiency) -- and is itself a sharp benchmark insight: \emph{TUAB frozen-probe is easy for simple power features -- in our consistent linear-probe setup, several published ``foundation-model'' frozen numbers barely clear this random baseline.}
  \end{alertblock}
\end{frame}

  \end{itemize}
\end{frame}

\section{Insight}
% ==========================================================

\begin{frame}{The result we pre-registered against: a clean null}
  \vspace{0.3em}
  3-seed mean balanced accuracy, $\{1,1000,10000\}$:
  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    \textbf{reg} $\backslash$ \textbf{space} & \textbf{ambient} & \textbf{tangent} \\
    \midrule
    VICReg & 0.814 & --- \\
    \rowcolor{vcolor!12} SIGReg & \textbf{0.819} (locked) & 0.820$^\dagger$ \\
    PEIRA  & 0.815 & 0.807 \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.5em}
  \begin{itemize}
    \item Every cell $\approx 0.82 \pm$ seed noise; per-cell ranges overlap heavily. \textbf{No effect of regulariser, no effect of space, no interaction.}
    \item Our hypothesis (\emph{geometry / PEIRA win}) is \textbf{falsified}. We \emph{locked} ambient SIGReg and kept the $2\times2$ as the published ablation.
  \end{itemize}
  \vspace{0.2em}
  {\scriptsize $^\dagger$ caveat: SIGReg's slice schedule is seeded per-step, not per-run, so its ``tightest variance'' is partly an artifact -- we do not lean on it (see appendix).}
\end{frame}

\begin{frame}{Why the naive tangent is mis-specified (the mechanism)}
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.54\textwidth}
    On the SPD manifold, the natural Gaussian is a \textbf{wrapped Gaussian} -- \emph{non-isotropic} by construction (de Surrel et al.\ 2025, Def.\ 4.1: $t\sim\mathcal N(\mu,\Sigma)$ with full $\Sigma$, not $\sigma^2 I$).

    \vspace{0.4em}
    SIGReg forces \emph{isotropy} ($\mathcal N(0,I)$) where the geometry induces a non-isotropic covariance -- it fights the structure that carries the signal.

    \column{0.46\textwidth}
    \begin{center}
    \footnotesize
    \begin{tabular}{lll}
      \toprule
       & \textbf{ours} & \textbf{principled} \\
      \midrule
      target & isotropic & wrapped, full $\Sigma$ \\
      base pt & identity & Fr\'echet mean \\
      \bottomrule
    \end{tabular}
    \end{center}
    \vspace{0.3em}
    \begin{block}{Honest framing}
      \footnotesize Future-work \emph{motivation}, not a validated mechanism: at $n{=}3$ the tangent effect is null, so our data \emph{cannot} confirm the wrapped-Gaussian prediction.
    \end{block}
  \end{columns}
  \vspace{0.3em}
  \begin{center}\footnotesize
    And the principled object is \emph{already} our Riemannian baseline ($\riem \ll 0.82$): the linear probe re-derives second-order tangent structure for free, so best case for tangent SSL is a \emph{tie}.
  \end{center}
\end{frame}

\begin{frame}{We \emph{measured} the geometry -- not just hid it}
  \vspace{0.4em}
  \begin{itemize}
    \item The probe reads the first-order \texttt{represent}; the SPD-tangent lives \emph{only} in the SSL loss. So we exposed the \textbf{learned tangent} to the probe directly.
    \item Learned-tangent probe: \textbf{0.79} BA (SIGReg-tangent 0.790 $>$ its random-tangent floor 0.776) -- a \emph{real but weak} second-order signal, \textbf{below} the first-order rep (0.82).
    \item TUAB rewards first-order power; the second-order tangent caps $\sim$0.78--0.80 (cf.\ MENDR 0.78--0.80, Riemann \riem). \textbf{The tangent is not broken -- it is capped by the task.}
  \end{itemize}
\end{frame}

\begin{frame}{What we claim -- and what we don't}
  \vspace{0.4em}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{\textcolor{ccolor!60!black}{DO claim}}
    \begin{itemize}\footnotesize
      \item Strongest among published \emph{frozen}/linear-probe TUAB baselines; inside the fine-tuned BA band.
      \item A clean \textbf{3-seed null} on geometry/PEIRA, with a geometric mechanism + precedent.

\section{Limitations \& Future Work}
% ==========================================================

\begin{frame}{Limitations \& Future Work}
  \vspace{0.6em}
  \begin{itemize}
    \item \textbf{Single dataset.} Specialization $\neq$ generality. The honest generality test is frozen cross-dataset transfer (TUEV/Sleep-EDF) -- started, but left as future work rather than over-claimed.
    \item \textbf{Underpowered null.} $n{=}3$ seeds; the inter-cell gaps ($\leq 0.013$) are below a plausible eval bootstrap CI ($\pm0.02$--$0.03$, not yet computed). The honest statement is ``\emph{benchmark-resolution-limited}'', not ``geometry is dead''.
    \item \textbf{The principled fix.} A wrapped-Gaussian SIGReg on the \emph{channel} covariance, AIRM tangent at the running Fr\'echet mean -- $\sim$40 LOC $+$ one re-pretrain, with real \texttt{eigh}-backward numerical risk.
    \item \textbf{Where geometry should pay off.} Robustness, calibration, and transfer under montage/threshold shift -- \emph{not} a balanced-accuracy race.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix: encoder block + readouts}
  \footnotesize
  Each backbone block: \texttt{Conv1d}(kernel 7, stride 2) $\to$ \texttt{BatchNorm} $\to$ \texttt{GELU}. Widths $\nchan\!\to\!64\!\to\!128\!\to\!128\!\to\!\dmodel$; a $1{\times}1$ head fixes the channel dim. Time $T:\winsamp\!\to\!\tprime$.
  \vspace{0.4em}
  \begin{itemize}
    \item \texttt{represent(x)} $=$ time-pooled feature map $\in\mathbb{R}^{\dmodel}$ -- the \textbf{only} thing the probe ever reads.
    \item \texttt{cov\_features(x)} $=$ a $1{\times}1$ projection to $\dcov$ channels, fed to the tangent map. $\dcov < \tprime$ keeps the covariance full-rank.
    \item Projector for SSL: \texttt{512-512-256} MLP (Linear + BN + GELU), discarded after pretraining.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix: the SPD-tangent construction}
  \footnotesize
  From the feature map $F\in\mathbb{R}^{\dcov\times T'}$ (centred over time):
  \begin{align*}
    C &= \tfrac{1}{T'-1}\,F F^\top + \varepsilon I \in \mathrm{SPD}(\dcov) &&\text{(temporal covariance)}\\
    \log_m C &= U \,\mathrm{diag}(\log \lambda_i)\, U^\top &&\text{(via eigendecomposition)}\\
    \mathbf{t} &= \mathrm{vec}_{\sqrt 2}(\log_m C) \in \mathbb{R}^{\dcov(\dcov+1)/2} &&\text{(off-diagonals scaled by }\sqrt2)
  \end{align*}
  \vspace{0.2em}
  \begin{itemize}
    \item The $\sqrt2$ weighting makes the Euclidean norm of $\mathbf t$ equal the Frobenius norm of $\log_m C$ -- the correct isometry for both LE and AIRM \emph{at the identity}.
    \item At the identity base point, AIRM-$\log$ $\equiv$ Log-Euclidean: the metric is \emph{not} an independent axis. Mis-specification $=$ isotropic target $+$ identity base point (two axes, not three).
    \item \texttt{eigh}-backward numerics: $\varepsilon$-clamped eigenvalues, small $\dcov$, float-safe; verified finite on GPU in the smoke test.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix: collapse diagnostics}
  \footnotesize
  On a batch of embeddings $Z\in\mathbb{R}^{N\times D}$ (mean-centred), with singular values $s_i$ and $p_i = s_i/\sum_j s_j$:
  \[ \text{eff.\ rank} = \exp\!\Big(-\textstyle\sum_i p_i \log p_i\Big), \quad \text{feat.\ std} = \overline{\mathrm{std}(Z_{:,d})}, \quad \text{off-diag} = \tfrac{\|\,\mathrm{Cov}(Z) - \mathrm{diag}\,\|_F}{\sqrt{D(D-1)}}. \]
  \begin{itemize}
    \item Collapse $\Rightarrow$ eff.\ rank $\to 1$, feat.\ std $\to 0$. Healthy SIGReg-ambient run: eff.\ rank $27\to65$ over training.
    \item For PEIRA, the loss value is a \emph{surrogate-gradient} number (can be negative, non-monotone). Only $\mathrm{tr}_P$ and eff.\ rank are valid progress signals.
    \item Per-cell non-degeneracy was audited before averaging -- tangent-PEIRA (0.807) was confirmed healthy, not a partial collapse.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix: evaluation protocol}
  \footnotesize
  \begin{itemize}
    \item \textbf{Feature extraction}: 16 evenly-spaced \winsec~s windows per recording $\to$ pooled \texttt{represent} $\to$ averaged to a recording-level vector.
    \item \textbf{Probe}: \texttt{StandardScaler} $+$ logistic regression on the frozen features. \textbf{Frozen encoder, no fine-tuning.}
    \item \textbf{Split}: train recordings fit the probe; the \nrecev{}-recording eval (held-out patients) is scored \textbf{once}. Probe hyper-parameters are fixed \emph{a priori} (no search); a patient-disjoint dev-split for any future selection is specified but not yet wired in.
    \item \textbf{Metrics}: balanced accuracy (primary) $+$ AUROC, on the full \nrectr/\nrecev{} split.
    \item \textbf{Floor}: an untrained random encoder, same probe, gives the $\sim$\floorv{} honest denominator.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix: caveats we disclose to the jury}
  \footnotesize
  \begin{itemize}
    \item \textbf{Report \ba{}, not \babest{}.} The single-seed \babest{} (SIGReg-ambient, seed 1) was the \emph{top} of that cell's range; the 3-seed mean is \ba{}.
    \item \textbf{SIGReg slice-seed bug (found, now fixed).} The reported 3-seed SIGReg numbers were produced by code that seeded \texttt{BCS} slice directions from a per-step counter (run-seed-independent), so the 3 seeds shared a byte-identical slice schedule -- the error bars \emph{understate} true variance. The code is now fixed (\texttt{losses.py} draws slices from the run-seeded RNG); the reported numbers predate the fix and were deliberately not re-run.
    \item \textbf{PEIRA tangent (0.807) is not evidence against wrapped-Gaussian.} PEIRA is distribution-free $\Rightarrow$ immune to the isotropic-target mis-specification $\Rightarrow$ it never tests that hypothesis. $0.807$ vs $0.815$ is within noise.
    \item \textbf{Bootstrap CI (illustrative, not yet run).} A bootstrap over the \nrecev{} eval recordings would plausibly give a $\pm0.02$--$0.03$ CI -- wider than the max inter-cell gap (0.013) -- supporting ``benchmark-resolution-limited''. The resampling is not yet committed.
    \item \textbf{Preprocessing differs} from the published FM rows -- the head-to-head is indicative, not a controlled re-run of their pipelines.
  \end{itemize}
\end{frame}
