\section{Contribution}
% ==========================================================

\begin{frame}{Our contribution, in one slide}
  \small
  \begin{enumerate}
    \item \textbf{A strong frozen baseline.} SIGReg-ambient JEPA matches the LeJEPA/Laya recipe's value on our (pipeline-helped) setup -- not a controlled head-to-head.
    \item \textbf{Nothing beats it.} Ambient/tangent SIGReg \& PEIRA, Graph-JEPA, Fourier-JEPA -- none beats ambient SIGReg in balanced accuracy (gaps $\leq 0.013$).
    \item \textbf{Latent geometry is the contribution.} Ours separates normal/abnormal $\sim$2$\times$ better than random, visualised with \textbf{de Surrel's AIRM Riemannian t-SNE} (the right metric for SPD/EEG covariances). Tangent sharpens the \emph{geometry} (best cov-latent silhouette), not the task.
    \item \textbf{Rigor over spin.} The limits we report -- \emph{decodable $\ne$ clustered}, sustained vs transient, AIRM edge within noise. An honest negative beats a fragile positive.
  \end{enumerate}
\end{frame}

% ==========================================================
\section{Motivation}
% ==========================================================

\begin{frame}{The question we actually asked}
  \vfill
  \begin{itemize}
    \item \textbf{EEG foundation models} are everywhere -- but most report \emph{fine-tuned} numbers. What survives when you \textbf{freeze} the encoder?
    \item \textbf{Self-supervised JEPA} avoids collapse with an \emph{anti-collapse regulariser}. The EEG covariance lives on a curved \textbf{SPD manifold}, not in flat Euclidean space.
    \item \textbf{Our pre-registered hypothesis}: making anti-collapse \emph{geometry-aware} (act in the SPD \textbf{tangent}) and \emph{distribution-free} (\textbf{PEIRA}) beats a plain ambient baseline.
    \item \textbf{What we ship}: a controlled study with an \emph{honest} answer -- a strong frozen probe, a clean negative result with a geometric mechanism, and label/data-efficiency curves.
  \end{itemize}
\end{frame}

\section{Data}
% ==========================================================

\begin{frame}{Data: TUH Abnormal EEG (TUAB)}
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{itemize}
      \item \textbf{Task}: binary \emph{normal} vs \emph{abnormal} EEG, recording level.
      \item \textbf{Signal}: \nchan{} channels @ \sfreq~Hz; we sample \winsec~s windows ($\winsamp$ samples), per-channel z-scored.
      \item \textbf{Standard split}: \nrectr{} train / \nrecev{} eval recordings, \textbf{patient-disjoint} (\npattr{} vs \npatev{} patients, overlap $=0$ -- verified).
      \item \textbf{Pretraining}: SSL reads only \emph{unlabeled} train recordings; labels enter at the probe alone.
    \end{itemize}
    \column{0.44\textwidth}
    \begin{block}{Why we don't chase accuracy}
      \footnotesize
      TUAB labels come from clinical reports and are known to be noisy; we are not aware of a published label-noise / Bayes-ceiling estimate.
      \textbf{Sub-1-point gaps without multi-seed CIs are not meaningful} -- so we report
      \emph{balanced accuracy on the full split}, 3 seeds, with error bars.
    \end{block}
  \end{columns}
\end{frame}

\end{frame}

    \textbf{TUEV -- 6 event types (decodable, not clustered)}\\[-0.1em]
    \includegraphics[width=\textwidth]{riemann_latent_tuev_annot.png}

  \begin{columns}[T]
    \column{0.40\textwidth}\centering
    \begin{tabular}{lcc}
      \toprule
      task (\#cls) & sil$_{\text{Euc}}$ & sil$_{\text{AIRM}}$ \\
      \midrule
      TUAB (2, abnormal)  & 0.049 & \textbf{0.052} \\

      TUEV (6, events)    & $-$0.003 & $-$0.004 \\
      \bottomrule
    \end{tabular}
    \column{0.60\textwidth}
    \begin{itemize}
      \item Clusters \textbf{sustained} phenomena (abnormality, seizures), not \textbf{transient} events -- TUEV's sub-second events are diluted by the 10\,s window.
      \item \textbf{Silhouette $\ne$ decodability}: TUEV stays linearly decodable (probe BA ${\sim}0.40$, 1st- \emph{and} 2nd-order, $\gg$ chance 0.17); clustering fails, not the features.
      \item AIRM $\ge$ Euclidean \emph{where there is cluster structure} (TUAB, TUSZ); at TUEV's ${\approx}0$ it is within noise.
      \item SIGReg (ours) $>$ random-init on every task, in both decodability and clustering.
    \end{itemize}
  \end{columns}
  \vspace{0.2em}
  {\tiny $^\dagger$ A \emph{label-optimal} projection (shrinkage-LDA, train$\to$eval) still does not cluster the classes (sup.\ silhouette ${\approx}0$) -- confirming \emph{decodable $\ne$ clustered}; the signal is a linear gradient, not separated modes.}
\end{frame}

\begin{frame}{Caveats we disclose to the jury}
  \footnotesize
  \begin{itemize}
    \item \textbf{Report \ba{}, not \babest{}.} The single-seed \babest{} (SIGReg-ambient, seed 1) was the \emph{top} of that cell's range; the 3-seed mean is \ba{}.
    \item \textbf{SIGReg slice-seed bug (found, fixed).} \texttt{BCS} seeded slice directions per-step, so the 3 SIGReg seeds shared one slice schedule -- error bars \emph{understate} variance. Now fixed (run-seeded RNG); the reported numbers predate the fix, not re-run.
    \item \textbf{PEIRA tangent (0.807) is not evidence against wrapped-Gaussian.} PEIRA is distribution-free $\Rightarrow$ immune to the isotropic-target mis-specification $\Rightarrow$ it never tests that hypothesis. $0.807$ vs $0.815$ is within noise.
    \item \textbf{Bootstrap CI (illustrative, not yet run).} A bootstrap over the \nrecev{} eval recordings would plausibly give a $\pm0.02$--$0.03$ CI -- wider than the max inter-cell gap (0.013) -- supporting ``benchmark-resolution-limited''. The resampling is not yet committed.
    \item \textbf{Preprocessing differs} from the published FM rows -- the head-to-head is indicative, not a controlled re-run of their pipelines.
  \end{itemize}
\end{frame}

\section{Limitations \& Future Work}
% ==========================================================

\begin{frame}{Limitations \& Future Work}
  \vspace{0.6em}
  \begin{itemize}
    \item \textbf{Single dataset.} Specialization $\neq$ generality. The honest generality test is frozen cross-dataset transfer (TUEV/Sleep-EDF) -- started, but left as future work rather than over-claimed.
    \item \textbf{Underpowered null.} $n{=}3$ seeds; the inter-cell gaps ($\leq 0.013$) are below a plausible eval bootstrap CI ($\pm0.02$--$0.03$, not yet computed). The honest statement is ``\emph{benchmark-resolution-limited}'', not ``geometry is dead''.
    \item \textbf{The principled fix.} A wrapped-Gaussian SIGReg on the \emph{channel} covariance, AIRM tangent at the running Fr\'echet mean -- $\sim$40 LOC $+$ one re-pretrain, with real \texttt{eigh}-backward numerical risk.
    \item \textbf{Where geometry should pay off.} Robustness, calibration, and transfer under montage/threshold shift -- \emph{not} a balanced-accuracy race.
  \end{itemize}
\end{frame}

\section{Conclusion}
% ==========================================================

\begin{frame}{Conclusion}
  \vspace{0.8em}
  \begin{itemize}
    \item A \textbf{controlled frozen-transfer study}, not a leaderboard entry.
    \item \textbf{A strong frozen probe} (\ba{} BA, $\sim$\aurocv{} AUROC) that holds where published FMs collapse when frozen -- with the random-encoder floor honestly disclosed.
    \item \textbf{A clean, pre-registered negative result}: \emph{where} anti-collapse acts (ambient vs SPD-tangent) and \emph{which} regulariser do not change frozen-probe accuracy on TUAB -- with a geometric reason why.
    \item \textbf{The null is multi-axis}: geometry/PEIRA help neither accuracy, nor calibration, nor robustness (3 seeds) -- and it holds even \emph{general}-pretrained (TUSZ $\to$ frozen TUAB, 0.814).
    \item \textbf{Honesty as the headline}: an honest negative beats a fragile positive.
  \end{itemize}
